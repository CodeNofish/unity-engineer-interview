给你无向 连通 图中一个节点的引用，请你返回该图的 深拷贝（克隆）。

图中的每个节点都包含它的值 val（int） 和其邻居的列表（list[Node]）。

#### 无向连通图 深度优先遍历

```C#
/*
// Definition for a Node.
public class Node {
    public int val;
    public IList<Node> neighbors;

    public Node() {
        val = 0;
        neighbors = new List<Node>();
    }

    public Node(int _val) {
        val = _val;
        neighbors = new List<Node>();
    }

    public Node(int _val, List<Node> _neighbors) {
        val = _val;
        neighbors = _neighbors;
    }
}
*/

public class Solution {
    internal Dictionary<Node, Node> visited = new Dictionary<Node, Node>();

    public Node CloneGraph(Node node) {
        if (node == null) {
            return node;
        }
        // 如果已经访问过了
        if (visited.ContainsKey(node)) {
            return visited[node];
        }

        // 克隆当前节点
        Node cloneNode = new Node(node.val, new List<Node>());
        // 原节点 => 克隆后节点
        visited.Add(node, cloneNode);

        foreach (Node nerighbor in node.neighbors) {
            cloneNode.neighbors.Add(CloneGraph(nerighbor));
        }
        return cloneNode;
    }
}
```
