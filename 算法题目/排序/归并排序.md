
归并排序是一种基于​**​分治法​**​的高效排序算法，核心思想是将数组不断拆分为子数组进行排序，再合并有序子数组。以下是详细解释和C#实现：

**算法步骤**
**分解​**​：将数组递归拆分为左右两半，直到子数组长度为1（天然有序）。
​**​合并​**​：将两个有序子数组合并为一个有序数组（通过比较元素大小依次填充）。

```C#
namespace AlgorithmPlayground;

public class MergeSort {
  public static void Sort(int[] array) {
    if (array == null || array.Length <= 1)
      return;
    // 临时数组
    int[] temp = new int[array.Length];
    Sort(array, 0, array.Length - 1, temp);
  }
  
  private static void Sort(int[] array, int left, int right, int[] temp) {
    // 长度为1 天然有序
    if (left >= right)
      return;
    // 分割点
    int mid = left + (right - left) / 2;
    // 排序左边
    Sort(array, left, mid, temp);
    // 排序右边
    Sort(array, mid + 1, right, temp);
    // 合并数组
    Merge(array, left, mid, right, temp);
  }

  private static void Merge(int[] array, int left, int mid, int right, int[] temp) {
    // 左右子数组的索引
    int i = left;
    int j = mid + 1;
    // 临时数组索引
    int k = 0;

    // 合并两个有序数组
    while (i <= mid && j <= right) {
      if (array[i] <= array[j]) 
        temp[k++] = array[i++];
      else
        temp[k++] = array[j++];
    }
    while (i <= mid) temp[k++] = array[i++];
    while (j <= right) temp[k++] = array[j++];
    
    // 将临时数组的数据 复制回原数组
    Array.Copy(temp, 0, array, left, k);
  }
}
```
